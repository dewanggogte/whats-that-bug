> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add two daily challenges (Bugs 101 + All Bugs) with progressive crop-reveal, autocomplete guessing, streaks, and shareable results.

**Architecture:** Static pipeline — a build script generates cropped images and a manifest JSON. The game client is pure vanilla JS in the existing Astro static site. No backend changes. Streaks/state in localStorage. Events logged via existing Google Sheets webhook.

**Tech Stack:** Astro 4, vanilla JS, Sharp (build-time image processing), Vitest, existing CSS custom properties.

```js
// src/scripts/daily-engine.js
/**
 * Daily challenge engine — pure logic, no DOM dependencies.
 * Handles timezone, streaks, guess validation, and state persistence.
 */

/** The first daily challenge date. Challenge #1. */
const EPOCH = '2026-04-07';

/**
 * Get today's date in Eastern Time as YYYY-MM-DD.
 * Daily challenges reset at midnight ET.
 */
export function getTodayET() {
  const now = new Date();
  const et = new Date(now.toLocaleString('en-US', { timeZone: 'America/New_York' }));
  const y = et.getFullYear();
  const m = String(et.getMonth() + 1).padStart(2, '0');
  const d = String(et.getDate()).padStart(2, '0');
  return `${y}-${m}-${d}`;
}

/**
 * Get the challenge number for a given date.
 * Day 1 = EPOCH date. Increments by 1 per day.
 */
export function getChallengeNumber(dateStr) {
  const epoch = new Date(EPOCH + 'T00:00:00');
  const target = new Date(dateStr + 'T00:00:00');
  const diffMs = target - epoch;
  return Math.floor(diffMs / (24 * 60 * 60 * 1000)) + 1;
}
```

```js
/**
 * Walk backward from today counting consecutive days played/won.
 * @param {Object<string, {solved: boolean, guesses: number}>} history
 * @param {string} today — YYYY-MM-DD
 * @returns {{ playStreak: number, winStreak: number }}
 */
export function calculateStreaks(history, today) {
  let playStreak = 0;
  let winStreak = 0;
  let winBroken = false;

  let current = new Date(today + 'T00:00:00');
  while (true) {
    const key = current.toISOString().slice(0, 10);
    const entry = history[key];
    if (!entry) break;

    playStreak++;
    if (entry.solved && !winBroken) {
      winStreak++;
    } else {
      winBroken = true;
    }

    current.setDate(current.getDate() - 1);
  }

  return { playStreak, winStreak };
}
```

```js
/**
 * Load today's daily challenge result from localStorage.
 * @param {'bugs101'|'allbugs'} mode
 * @param {string} dateStr — YYYY-MM-DD
 * @returns {null|{solved: boolean, guesses: number, answer: string}}
 */
export function loadDailyState(mode, dateStr) {
  try {
    const raw = localStorage.getItem(`daily_${mode}_history`);
    if (!raw) return null;
    const history = JSON.parse(raw);
    return history[dateStr] || null;
  } catch {
    return null;
  }
}

/**
 * Save a daily challenge result to localStorage.
 * @param {'bugs101'|'allbugs'} mode
 * @param {string} dateStr — YYYY-MM-DD
 * @param {{solved: boolean, guesses: number, answer: string}} result
 */
export function saveDailyResult(mode, dateStr, result) {
  const key = `daily_${mode}_history`;
  let history = {};
  try {
    const raw = localStorage.getItem(key);
    if (raw) history = JSON.parse(raw);
  } catch { /* fresh start */ }
  history[dateStr] = result;
  localStorage.setItem(key, JSON.stringify(history));
}

/**
 * Load full history object from localStorage.
 * @param {'bugs101'|'allbugs'} mode
 * @returns {Object<string, {solved: boolean, guesses: number}>}
 */
export function loadHistory(mode) {
  try {
    const raw = localStorage.getItem(`daily_${mode}_history`);
    return raw ? JSON.parse(raw) : {};
  } catch {
    return {};
  }
}

/**
 * Get countdown to next daily reset (midnight ET).
 * @returns {{ hours: number, minutes: number }}
 */
export function getCountdownToReset() {
  const now = new Date();
  const etStr = now.toLocaleString('en-US', { timeZone: 'America/New_York' });
  const et = new Date(etStr);

  const midnight = new Date(et);
  midnight.setDate(midnight.getDate() + 1);
  midnight.setHours(0, 0, 0, 0);

  const diffMs = midnight - et;
  const hours = Math.floor(diffMs / (1000 * 60 * 60));
  const minutes = Math.floor((diffMs % (1000 * 60 * 60)) / (1000 * 60));
  return { hours, minutes };
}
```

```js
// src/scripts/daily-share.js
/**
 * Share text generation for daily challenges.
 */

/**
 * Get a flavor line based on performance.
 * @param {number} guessCount — which guess they solved on (1-based)
 * @param {boolean} solved
 * @returns {string}
 */
export function getDailyFlavor(guessCount, solved) {
  if (!solved) return 'Better luck tomorrow!';
  if (guessCount === 1) return 'Incredible! First try! 🏆';
  if (guessCount === 2) return 'Impressive eye! 🔬';
  if (guessCount <= 4) return 'Well spotted!';
  return 'Just in time!';
}

/**
 * Generate shareable text for a daily challenge result.
 * @param {{
 *   mode: 'bugs101'|'allbugs',
 *   challengeNumber: number,
 *   solved: boolean,
 *   guesses: boolean[],   // array of correct/incorrect per guess
 *   maxGuesses: number,
 * }} opts
 * @returns {string}
 */
export function generateDailyShareText({ mode, challengeNumber, solved, guesses, maxGuesses }) {
  const modeLabel = mode === 'bugs101' ? 'Bugs 101' : 'All Bugs';
  const guessCount = guesses.length;
  const scoreLabel = solved ? `${guessCount}/${maxGuesses}` : `X/${maxGuesses}`;

  const grid = guesses.map(correct => correct ? '🟩' : '🟥').join('');
  const remaining = solved ? '⬜'.repeat(maxGuesses - guessCount) : '';

  const flavor = getDailyFlavor(guessCount, solved);

  return [
    `🪲 What's That Bug`,
    modeLabel,
    `Daily #${challengeNumber} — ${scoreLabel}`,
    grid + remaining,
    '',
    flavor,
    '',
    'https://dewanggogte.com/games/bugs/daily?ref=share',
  ].join('\n');
}
```

```js
  { level: 3, size: 0.35 },
  { level: 4, size: 0.50 },
  { level: 5, size: 0.70 },
  { level: 6, size: 0.90 },
];

const CROP_OUTPUT_WIDTH = 800;
const CROP_OUTPUT_HEIGHT = 600;

// --- Load observations ---
const observations = JSON.parse(readFileSync(join(DATA_DIR, 'observations.json'), 'utf8'));

// --- Load used observations (to avoid repeats) ---
function loadUsed() {
  if (existsSync(USED_PATH)) {
    return new Set(JSON.parse(readFileSync(USED_PATH, 'utf8')));
  }
  return new Set();
}

function saveUsed(usedSet) {
  writeFileSync(USED_PATH, JSON.stringify([...usedSet], null, 2));
}

// --- Load existing manifest ---
function loadManifest() {
  if (existsSync(MANIFEST_PATH)) {
    return JSON.parse(readFileSync(MANIFEST_PATH, 'utf8'));
  }
  return { challenges: [] };
}

// --- Candidate selection ---
function selectCandidates(count, usedSet, preferredOrders = null) {
  // Filter for observations that:
  // 1. Haven't been used before
  // 2. Have photos (photo_url exists)
  const candidates = observations.filter((obs, idx) => {
    if (usedSet.has(obs.id)) return false;
    if (!obs.photo_url) return false;
    return true;
  });

  // Shuffle for randomness
  const shuffled = candidates.sort(() => Math.random() - 0.5);

  // If we want order diversity (bugs101), pick one per order
  if (preferredOrders) {
    const byOrder = {};
    for (const obs of shuffled) {
      const order = obs.taxon?.order;
      if (order && !byOrder[order]) {
        byOrder[order] = obs;
      }
    }
    const ordered = Object.values(byOrder).sort(() => Math.random() - 0.5);
    return ordered.slice(0, count);
  }

  return shuffled.slice(0, count);
}

// --- Download original image from iNaturalist ---
async function downloadOriginal(photoUrl) {
  // Convert medium.jpeg URL to original.jpeg
  const originalUrl = photoUrl.replace('/medium.', '/original.');
  console.log(`  Downloading: ${originalUrl}`);

  const response = await fetch(originalUrl);
  if (!response.ok) {
    // Fall back to large if original unavailable
    const largeUrl = photoUrl.replace('/medium.', '/large.');
    console.log(`  Falling back to large: ${largeUrl}`);
    const fallbackRes = await fetch(largeUrl);
    if (!fallbackRes.ok) throw new Error(`Failed to download image: ${largeUrl}`);
    return Buffer.from(await fallbackRes.arrayBuffer());
  }
  return Buffer.from(await response.arrayBuffer());
}

// --- Check image resolution ---
async function getImageDimensions(buffer) {
  const metadata = await sharp(buffer).metadata();
  return { width: metadata.width, height: metadata.height };
}

// --- Generate crops ---
async function generateCrops(imageBuffer, cropDefs, outputDir, prefix) {
  const { width, height } = await getImageDimensions(imageBuffer);
  const centerX = width / 2;
  const centerY = height / 2;

  const crops = [];

  for (const { level, size } of cropDefs) {
    const cropW = Math.round(width * size);
    const cropH = Math.round(height * size);
    const left = Math.round(centerX - cropW / 2);
    const top = Math.round(centerY - cropH / 2);

    const filename = `${prefix}_${level}.jpg`;
    const outputPath = join(outputDir, filename);

    await sharp(imageBuffer)
      .extract({ left, top, width: cropW, height: cropH })
      .resize(CROP_OUTPUT_WIDTH, CROP_OUTPUT_HEIGHT, { fit: 'cover' })
      .jpeg({ quality: 85 })
      .toFile(outputPath);

    crops.push(`daily/${outputDir.split('daily/')[1]}/${filename}`);
  }

  // Full reveal image
  const revealFilename = `${prefix}_full.jpg`;
  const revealPath = join(outputDir, revealFilename);
  await sharp(imageBuffer)
    .resize(CROP_OUTPUT_WIDTH * 2, CROP_OUTPUT_HEIGHT * 2, { fit: 'inside', withoutEnlargement: true })
    .jpeg({ quality: 90 })
    .toFile(revealPath);

  const revealRelPath = `daily/${outputDir.split('daily/')[1]}/${revealFilename}`;

  return { crops, reveal: revealRelPath };
}

// --- Bugs 101 name mapping (mirrors game-ui.js getBugs101Name) ---
const BEE_FAMILIES = ['Apidae', 'Megachilidae', 'Halictidae', 'Andrenidae', 'Colletidae'];
const ANT_FAMILIES = ['Formicidae', 'Mutillidae'];
const BUTTERFLY_FAMILIES = ['Nymphalidae', 'Papilionidae', 'Pieridae', 'Lycaenidae', 'Riodinidae', 'Hesperiidae'];
const CRICKET_FAMILIES = ['Gryllidae', 'Rhaphidophoridae', 'Anostostomatidae', 'Tettigoniidae'];
const DAMSELFLY_FAMILIES = ['Coenagrionidae', 'Calopterygidae', 'Lestidae', 'Platycnemididae', 'Platystictidae'];
const CICADA_FAMILIES = ['Cicadidae'];
const STINK_BUG_FAMILIES = ['Pentatomidae', 'Scutelleridae', 'Acanthosomatidae', 'Cydnidae', 'Tessaratomidae'];
const PLANTHOPPER_FAMILIES = ['Fulgoridae', 'Flatidae', 'Membracidae', 'Ischnorhinidae'];
const APHID_FAMILIES = ['Aphididae', 'Eriococcidae'];
const WATER_BUG_FAMILIES = ['Nepidae', 'Notonectidae', 'Belostomatidae'];

function getBugs101Name(taxon) {
  if (taxon.order === 'Hymenoptera') {
    if (BEE_FAMILIES.includes(taxon.family)) return 'Bee';
    if (ANT_FAMILIES.includes(taxon.family)) return 'Ant';
    return 'Wasp';
  }
  if (taxon.order === 'Lepidoptera') {
    return BUTTERFLY_FAMILIES.includes(taxon.family) ? 'Butterfly' : 'Moth';
  }
  if (taxon.order === 'Orthoptera') {
    if (CRICKET_FAMILIES.includes(taxon.family)) return 'Cricket';
    return 'Grasshopper';
  }
  if (taxon.order === 'Odonata') {
    return DAMSELFLY_FAMILIES.includes(taxon.family) ? 'Damselfly' : 'Dragonfly';
  }
  if (taxon.order === 'Hemiptera') {
    if (CICADA_FAMILIES.includes(taxon.family)) return 'Cicada';
    if (STINK_BUG_FAMILIES.includes(taxon.family)) return 'Stink Bug';
    if (PLANTHOPPER_FAMILIES.includes(taxon.family)) return 'Planthopper';
    if (APHID_FAMILIES.includes(taxon.family)) return 'Aphid';
    if (WATER_BUG_FAMILIES.includes(taxon.family)) return 'Water Bug';
    return 'True Bug';
  }
  const names = {
    'Coleoptera': 'Beetle', 'Ixodida': 'Tick', 'Araneae': 'Spider',
    'Scorpiones': 'Scorpion', 'Opiliones': 'Harvestman', 'Mantodea': 'Mantis',
    'Diptera': 'Fly', 'Phasmida': 'Stick Insect', 'Neuroptera': 'Lacewing',
    'Blattodea': 'Cockroach', 'Dermaptera': 'Earwig', 'Ephemeroptera': 'Mayfly',
    'Trichoptera': 'Caddisfly',
  };
  return names[taxon.order] || taxon.order_common || taxon.order;
}

// --- Main pipeline ---
async function main() {
  console.log(`Generating ${DAYS} daily challenges starting from ${START_DATE}`);

  const usedSet = loadUsed();
  const manifest = loadManifest();

  // Dates already in manifest
  const existingDates = new Set(manifest.challenges.map(c => c.date));

  // Generate dates
  const dates = [];
  const start = new Date(START_DATE + 'T00:00:00');
  for (let i = 0; i < DAYS; i++) {
    const d = new Date(start);
    d.setDate(d.getDate() + i);
    const dateStr = d.toISOString().slice(0, 10);
    if (!existingDates.has(dateStr)) {
      dates.push(dateStr);
    }
  }

  if (dates.length === 0) {
    console.log('All dates already have challenges. Nothing to generate.');
    return;
  }

  console.log(`Generating for ${dates.length} new dates: ${dates.join(', ')}`);

  // Select candidates — need 2 per day (one bugs101, one allbugs)
  const bugs101Candidates = selectCandidates(dates.length, usedSet, true);
  const allbugsCandidates = selectCandidates(dates.length, usedSet);

  if (bugs101Candidates.length < dates.length || allbugsCandidates.length < dates.length) {
    console.error(`Not enough candidates! Need ${dates.length} per mode.`);
    console.error(`  bugs101: ${bugs101Candidates.length}, allbugs: ${allbugsCandidates.length}`);
    process.exit(1);
  }

  const MIN_DIMENSION = 1500;

  for (let i = 0; i < dates.length; i++) {
    const dateStr = dates[i];
    const challengeNumber = Math.floor((new Date(dateStr + 'T00:00:00') - new Date(EPOCH + 'T00:00:00')) / (24 * 60 * 60 * 1000)) + 1;

    console.log(`\n--- ${dateStr} (Challenge #${challengeNumber}) ---`);

    const dateDir = join(DAILY_DIR, dateStr);
    mkdirSync(dateDir, { recursive: true });

    // --- Bugs 101 ---
    const b101Obs = bugs101Candidates[i];
    console.log(`  Bugs 101: ${b101Obs.taxon.common_name} (${b101Obs.taxon.species})`);

    let b101Buffer;
    try {
      b101Buffer = await downloadOriginal(b101Obs.photo_url);
      const dims = await getImageDimensions(b101Buffer);
      console.log(`  Resolution: ${dims.width}x${dims.height}`);
      if (Math.min(dims.width, dims.height) < MIN_DIMENSION) {
        console.warn(`  WARNING: Image below ${MIN_DIMENSION}px minimum (${Math.min(dims.width, dims.height)}px). Crops may be low quality.`);
      }
    } catch (err) {
      console.error(`  Failed to download bugs101 image: ${err.message}`);
      continue;
    }

    const b101Result = await generateCrops(b101Buffer, BUGS101_CROPS, dateDir, 'b101');

    // --- All Bugs ---
    const abObs = allbugsCandidates[i];
    console.log(`  All Bugs: ${abObs.taxon.common_name} (${abObs.taxon.species})`);

    let abBuffer;
    try {
      abBuffer = await downloadOriginal(abObs.photo_url);
      const dims = await getImageDimensions(abBuffer);
      console.log(`  Resolution: ${dims.width}x${dims.height}`);
      if (Math.min(dims.width, dims.height) < MIN_DIMENSION) {
        console.warn(`  WARNING: Image below ${MIN_DIMENSION}px minimum (${Math.min(dims.width, dims.height)}px). Crops may be low quality.`);
      }
    } catch (err) {
      console.error(`  Failed to download allbugs image: ${err.message}`);
      continue;
    }

    const abResult = await generateCrops(abBuffer, ALLBUGS_CROPS, dateDir, 'expert');

    // Add to manifest
    manifest.challenges.push({
      date: dateStr,
      number: challengeNumber,
      approved: false,
      bugs101: {
        observation_id: b101Obs.id,
        answer_order: b101Obs.taxon.order,
        answer_common: getBugs101Name(b101Obs.taxon),
        crops: b101Result.crops,
        reveal: b101Result.reveal,
        attribution: b101Obs.attribution,
        wikipedia_summary: b101Obs.wikipedia_summary || '',
        inat_url: b101Obs.inat_url || `https://www.inaturalist.org/observations/${b101Obs.id}`,
      },
      allbugs: {
        observation_id: abObs.id,
        answer_species: abObs.taxon.species,
        answer_common: abObs.taxon.common_name,
        crops: abResult.crops,
        reveal: abResult.reveal,
        attribution: abObs.attribution,
        wikipedia_summary: abObs.wikipedia_summary || '',
        inat_url: abObs.inat_url || `https://www.inaturalist.org/observations/${abObs.id}`,
      },
    });

    usedSet.add(b101Obs.id);
    usedSet.add(abObs.id);
  }

  // Sort manifest by date
  manifest.challenges.sort((a, b) => a.date.localeCompare(b.date));

  // Save
  writeFileSync(MANIFEST_PATH, JSON.stringify(manifest, null, 2));
  saveUsed(usedSet);

  console.log(`\nGenerated ${dates.length} daily challenges.`);
  console.log(`Manifest: ${MANIFEST_PATH}`);
  console.log(`\nIMPORTANT: Review challenges with scripts/review-daily.html before deploying.`);
  console.log(`All challenges have approved: false and must be approved before they go live.`);
}

main().catch(err => {
  console.error('Pipeline failed:', err);
  process.exit(1);
});
```

```js
// Bugs 101 autocomplete list — matches getBugs101Name in game-ui.js
const BUGS101_OPTIONS = [
  'Ant', 'Aphid', 'Bee', 'Beetle', 'Butterfly', 'Caddisfly', 'Cicada',

function escapeHTML(str) {
  return String(str)
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;')
    .replace(/'/g, '&#039;');
}

/**
 * Initialize the daily challenge. Called from daily/play.astro.
 */
export async function initDaily() {
  container = document.getElementById('daily-container');
  container.setAttribute('aria-live', 'polite');

  const params = new URLSearchParams(window.location.search);
  mode = params.get('mode') || 'bugs101';
  maxGuesses = mode === 'bugs101' ? 3 : 6;

  today = getTodayET();
  challengeNumber = getChallengeNumber(today);

  // Check if already played today
  const existingResult = loadDailyState(mode, today);
  if (existingResult) {
    // Load the challenge data for the reveal
    await loadChallenge();
    if (challenge) {
      solved = existingResult.solved;
      guesses = existingResult.guessHistory || [];
      gameOver = true;
      renderReveal();
    }
    return;
  }

  // Load challenge data
  const loaded = await loadChallenge();
  if (!loaded) return;

  // Generate session ID
  sessionId = crypto.randomUUID ? crypto.randomUUID() :
    'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, c => {
      const r = (Math.random() * 16) | 0;
      return (c === 'x' ? r : (r & 0x3) | 0x8).toString(16);
    });

  logDailyStart(sessionId, mode, today, challengeNumber);

  // Load species list for allbugs autocomplete
  if (mode === 'allbugs') {
    try {
      const obsRes = await fetch(`${base}/data/observations.json`);
      const observations = await obsRes.json();
      const speciesSet = new Map();
      for (const obs of observations) {
        if (obs.taxon?.species && !speciesSet.has(obs.taxon.species)) {
          speciesSet.set(obs.taxon.species, obs.taxon.common_name || obs.taxon.species);
        }
      }
      allSpeciesList = Array.from(speciesSet.entries())
        .map(([species, common]) => ({ species, common, label: `${common} (${species})` }))
        .sort((a, b) => a.common.localeCompare(b.common));
    } catch {
      console.warn('Failed to load species list for autocomplete');
    }
  }

  renderGame();
}

async function loadChallenge() {
  try {
    const res = await fetch(`${base}/data/daily/manifest.json`);
    if (!res.ok) throw new Error('Manifest not found');
    const manifest = await res.json();

    const todayChallenge = manifest.challenges.find(c => c.date === today && c.approved);
    if (!todayChallenge) {
      container.innerHTML = `<div class="container" style="text-align:center;padding-top:80px;">
        <h2>No challenge today</h2>
        <p class="subtitle">Check back tomorrow!</p>
        <a href="${base}/daily" style="color:var(--accent);">Back to Daily Hub</a>
      </div>`;
      return false;
    }

    challenge = todayChallenge;
    return true;
  } catch (err) {
    container.innerHTML = `<div class="container" style="text-align:center;padding-top:80px;">
      <p>Failed to load daily challenge.</p>
      <p style="color:var(--text-secondary);font-size:13px;">${escapeHTML(err.message)}</p>
    </div>`;
    return false;
  }
}

let selectedAnswer = '';
let highlightedIndex = -1;

function setupAutocomplete() {
  const input = document.getElementById('guess-input');
  const dropdown = document.getElementById('autocomplete');
  const submitBtn = document.getElementById('submit-btn');

  input.addEventListener('input', () => {
    const query = input.value.trim().toLowerCase();
    selectedAnswer = '';
    submitBtn.disabled = true;
    highlightedIndex = -1;

    if (query.length < 1) {
      dropdown.classList.remove('open');
      return;
    }

    let matches;
    if (mode === 'bugs101') {
      matches = BUGS101_OPTIONS
        .filter(name => name.toLowerCase().includes(query))
        .map(name => ({ label: name, value: name }));
    } else {
      matches = allSpeciesList
        .filter(s =>
          s.common.toLowerCase().includes(query) ||
          s.species.toLowerCase().includes(query)
        )
        .slice(0, 15)
        .map(s => ({ label: s.common, value: s.common, scientific: s.species }));
    }

    if (matches.length === 0) {
      dropdown.classList.remove('open');
      return;
    }

    dropdown.innerHTML = matches.map((m, i) => {
      const sci = m.scientific ? `<span class="scientific">${escapeHTML(m.scientific)}</span>` : '';
      return `<div class="daily-autocomplete-item" data-value="${escapeHTML(m.value)}" data-idx="${i}">${escapeHTML(m.label)}${sci}</div>`;
    }).join('');
    dropdown.classList.add('open');

    dropdown.querySelectorAll('.daily-autocomplete-item').forEach(item => {
      item.addEventListener('click', () => {
        selectedAnswer = item.getAttribute('data-value');
        input.value = selectedAnswer;
        dropdown.classList.remove('open');
        submitBtn.disabled = false;
      });
    });
  });

  // Keyboard navigation
  input.addEventListener('keydown', (e) => {
    const items = dropdown.querySelectorAll('.daily-autocomplete-item');
    if (e.key === 'ArrowDown') {
      e.preventDefault();
      highlightedIndex = Math.min(highlightedIndex + 1, items.length - 1);
      updateHighlight(items);
    } else if (e.key === 'ArrowUp') {
      e.preventDefault();
      highlightedIndex = Math.max(highlightedIndex - 1, 0);
      updateHighlight(items);
    } else if (e.key === 'Enter') {
      e.preventDefault();
      if (highlightedIndex >= 0 && items[highlightedIndex]) {
        items[highlightedIndex].click();
      } else if (selectedAnswer) {
        submitGuess();
      }
    }
  });

  // Close on outside click
  document.addEventListener('click', (e) => {
    if (!e.target.closest('.daily-input-wrapper')) {
      dropdown.classList.remove('open');
    }
  });
}

function updateHighlight(items) {
  items.forEach((item, i) => {
    item.classList.toggle('highlighted', i === highlightedIndex);
  });
  if (highlightedIndex >= 0 && items[highlightedIndex]) {
    selectedAnswer = items[highlightedIndex].getAttribute('data-value');
    document.getElementById('guess-input').value = selectedAnswer;
    document.getElementById('submit-btn').disabled = false;
  }
}

// ===== SUBMIT =====

function setupSubmit() {
  document.getElementById('submit-btn').addEventListener('click', submitGuess);
}

function submitGuess() {
  if (!selectedAnswer || gameOver) return;

  const answer = getAnswerForValidation();
  const result = validateGuess(selectedAnswer, answer);
  guesses.push({ answer: selectedAnswer, correct: result.correct });

  logDailyGuess(sessionId, guesses.length, selectedAnswer, result.correct, mode, today);

  if (result.correct) {
    solved = true;
    gameOver = true;
    finishGame();
  } else {
    currentGuess++;
    if (currentGuess >= maxGuesses) {
      gameOver = true;
      finishGame();
    } else {
      // Show next crop
      const crops = getCrops();
      const imgEl = document.querySelector('#daily-image img');
      if (imgEl) imgEl.src = `${base}/data/${crops[currentGuess]}`;
      const badge = document.querySelector('.daily-guess-badge');
      if (badge) badge.textContent = `${currentGuess + 1} / ${maxGuesses}`;

      renderHistoryStrip();
      renderWrongGuesses();

      // Reset input
      const input = document.getElementById('guess-input');
      input.value = '';
      selectedAnswer = '';
      document.getElementById('submit-btn').disabled = true;
      input.focus();
    }
  }
}

function finishGame() {
  const guessHistory = guesses.map(g => ({ answer: g.answer, correct: g.correct }));
  saveDailyResult(mode, today, {
    solved,
    guesses: guesses.length,
    answer: getAnswer(),
    guessHistory,
  });

  const history = loadHistory(mode);
  const streaks = calculateStreaks(history, today);

  logDailyComplete(sessionId, mode, solved, guesses.length, today, false, streaks.playStreak, streaks.winStreak);

  renderReveal();
}
```

### Task 10: Vercel Config — Cache Headers

**Files:**
- Modify: `vercel.json`

The manifest.json for daily challenges changes weekly, so it must NOT have the 1-year immutable cache applied to the other data JSON files. The crop images themselves can be immutably cached.

- [ ] **Step 1: Update vercel.json**

Replace the current `vercel.json` content:

```json
{
  "headers": [
    {
      "source": "/data/(observations|taxonomy|sets).json",
      "headers": [
        {
          "key": "Cache-Control",
          "value": "public, max-age=31536000, immutable"
        }
      ]
    },
    {
      "source": "/data/daily/manifest.json",
      "headers": [
        {
          "key": "Cache-Control",
          "value": "public, max-age=3600"
        }
      ]
    },
    {
      "source": "/data/daily/:date/:file",
      "headers": [
        {
          "key": "Cache-Control",
          "value": "public, max-age=31536000, immutable"
        }
      ]
    }
  ]
}
```

This gives:
- `observations.json`, `taxonomy.json`, `sets.json` → 1-year immutable (same as before)
- `daily/manifest.json` → 1-hour cache (refreshes when new challenges are deployed)
- `daily/2026-04-07/*.jpg` → 1-year immutable (crop images never change)

### Task 11: GitHub Actions — Weekly Cron

**Files:**
- Create: `.github/workflows/generate-daily.yml`

- [ ] **Step 1: Create the workflow file**

```bash
mkdir -p .github/workflows
```

```yaml
# .github/workflows/generate-daily.yml
name: Generate Daily Challenges

on:
  schedule:
    # Run every Sunday at 06:00 UTC (1:00 AM ET)
    - cron: '0 6 * * 0'
  workflow_dispatch:
    inputs:
      days:
        description: 'Number of days to generate'
        default: '7'

jobs:
  generate:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'

      - run: npm ci

      - name: Generate daily challenges
        run: node scripts/generate-daily.mjs --days ${{ github.event.inputs.days || '7' }}

      - name: Check for new challenges
        id: check
        run: |
          if git diff --quiet public/data/daily/; then
            echo "changed=false" >> $GITHUB_OUTPUT
          else
            echo "changed=true" >> $GITHUB_OUTPUT
          fi

      - name: Create PR with new challenges
        if: steps.check.outputs.changed == 'true'
        env:
          GH_TOKEN: ${{ secrets.GITHUB_TOKEN }}
        run: |
          BRANCH="daily/$(date +%Y-%m-%d)"
          git checkout -b "$BRANCH"
          git add public/data/daily/
          git commit -m "feat(daily): generate challenges for week of $(date +%Y-%m-%d)"
          git push -u origin "$BRANCH"
          gh pr create \
            --title "Daily challenges: week of $(date +%Y-%m-%d)" \
            --body "Auto-generated daily challenges. Review with \`scripts/review-daily.html\` before merging." \
            --label "daily-challenge"
```

This creates a PR instead of pushing directly, so you can review the challenges before they go live. The review tool (`scripts/review-daily.html`) is used to approve/reject before merging the PR.
